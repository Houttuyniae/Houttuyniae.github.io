Houttuyniae 2019-12-08 Nginx

# linux 下安装

# 1. 安装 gcc gcc-c++(如新环境,未安装请先安装)

yum install -y gcc gcc-c++
1

可以通过 gcc -v命令 查看是否安装

# 2. 安装 wget

yum -y install wget
1

可以通过 rpm -qa|grep "wget"命令 查看是否安装

用来下载 Nginx

# 3. 安装 PCRE 库

rewirte 模块需要用的

cd /usr/local/
wget http://jaist.dl.sourceforge.net/project/pcre/pcre/8.33/pcre-8.33.tar.gz
tar -zxvf pcre-8.33.tar.gz
cd pcre-8.33
./configure
make && make install

# 4. 安装 openssl 库

ssl 模块需要用的

cd /usr/local/
wget http://www.openssl.org/source/openssl-1.0.1j.tar.gz
tar -zxvf openssl-1.0.1j.tar.gz
cd openssl-1.0.1j
./config
make && make install

# 安装 Nginx

wget http://nginx.org/download/nginx-1.8.0.tar.gz
tar -zxvf nginx-1.8.0.tar.gz
cd nginx-1.8.0
./configure
make && make install

# 启动并测试

云服务器需要开放防火墙 80 端口的规则，否则无法访问到

# 关闭服务器防火墙

//临时关闭
systemctl stop firewalld
//禁止开机启动
systemctl disable firewalld
